90template <class Container>
91[[nodiscard]] std::string
92to_string(const Container &container, const std::string &delimiter = ", ",
93 const std::array<std::string, 2> &surround = {"[", "]"}) {
94 auto iter = std::cbegin(container);
95 const auto end = std::cend(container);
96 if (iter == end) {
97 return surround[0] + surround[1];
98 }
99 std::ostringstream strStream;
100 strStream << surround[0] << *iter;
101 for (++iter; iter != end; ++iter) {
102 strStream << delimiter << *iter;
103 }
104 strStream << surround[1];
105 return strStream.str();
106}

121template <class Container, class F>
122inline Container elementWisePairOp(const Container &lhs, const Container &rhs,
123 F binaryFunction) {
124 Container ret = lhs;
125 auto retIter = std::begin(ret);
126 auto lhsIter = std::cbegin(lhs);
127 const auto lhsEnd = std::cend(lhs);
128 auto rhsIter = std::cbegin(rhs);
129 const auto rhsEnd = std::cend(rhs);
130
131 for (; lhsIter != lhsEnd and rhsIter != rhsEnd;
132 ++lhsIter, ++rhsIter, ++retIter) {
133 *retIter = binaryFunction(*lhsIter, *rhsIter);
134 }
135
136 return ret;
137}
